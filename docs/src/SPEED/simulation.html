 297    def simulate(self):
 298        """
 299        Execute the simulation main process.
 300        """
 301
 302        while True:
 303
 304            # Process the SFC Requests that arrived in the simulation time (env.now)
 305            if self.env.now in self.environment['sfc_requests_arrival']:
 306                sfc_requests = self.environment['sfc_requests_arrival'][self.env.now]
 307
 308                # Execute the SPC Placement in a distributed fashion for each SFC Request.
 309                for sfc_request in sfc_requests:
 310
 311                    # Update all the aggregated data in the simulation
 312                    self.update_aggregated_data()
 313
 314                    try:
 315
 316                        # Select the zone manager
 317                        aux = self.select_zone_manager(
 318                            sfc_request=sfc_request
 319                        )
 320
 321                        zone_manager: Zone = aux['zone_manager']
 322
 323                        vnf_names = DistributedServiceHelper.get_vnf_names_from_sfc_request(
 324                            sfc_request=sfc_request
 325                        )
 326
 327                        # The zone that will manage this request
 328                        self.sfc_request_zone_manager[sfc_request.name] = zone_manager
 329
 330                        placement_timeout = self.default_placement_timeout
 331
 332                        if sfc_request.extra_parameter_is_defined('placement_timeout'):
 333                            placement_timeout = sfc_request.get_extra_parameter('placement_timeout')
 334
 335                        self.zdsm[zone_manager.name].add_sfc_request(
 336                            sfc_request=sfc_request,
 337                            placement_timeout=placement_timeout
 338                        )
 339
 340                        # Lot the selection zone will manage the SFC Request
 341                        self.distributed_service_log.add_event(
 342                            event=DistributedServiceLog.ZONE_MANAGER_SELECTED,
 343                            time=self.env.now,
 344                            sfc_request_name=sfc_request.name,
 345                            zone_manager_name=zone_manager.name
 346                        )
 347
 348                    except TypeError:
 349                        # The requested service cannot be placed, there is no zone to manage this request.
 350                        self.distributed_service_log.add_event(
 351                            event=DistributedServiceLog.FAIL,
 352                            time=self.env.now,
 353                            sfc_request_name=sfc_request.name,
 354                            zone_manager_name="Not Found"
 355                        )
 356                        continue
 357
 358                    # Zone manager executing the game
 359                    try:
 360                        self.env.process(
 361                            self.distributed_sfc_placement_process(
 362                                sfc_request=sfc_request,
 363                                zone=zone_manager,
 364                                vnf_names=vnf_names
 365                            )
 366                        )
 367                    except TypeError:
 368                        print("Simulation error")
 369
 370            for i, domain in self.environment['domains'].items():
 371                self.dec_sfc_instances_timeout(domain)
 372                """
 373                Decrement the timeout of the SFC Instances.
 374                """
 375
 376                self.inc_packet_delay(domain)
 377                """
 378                Increment the delay of the packet in execution.
 379                """
 380
 381            # Iterate over the SFC Requests and log when the plan for placement arrives to the manager zone.
 382            self.sfc_requests_vnfs_are_assigned_to_compute_zone()
 383
 384            # Do the simulation tick of 1ms
 385            yield self.env.timeout(1)

 387    def distributed_sfc_placement_process(self, sfc_request: SFCRequest, zone: Zone, vnf_names: List, timeout: int = 0):
 388        """
 389        Execute the game in each zone.
 390
 391        :param sfc_request: The service requested
 392        :param zone: The zone where the game is executed
 393        :param vnf_names: The name of the VNFs.
 394        :param timeout: The delay between the distributed service component in the parent until the child
 395        zone component.
 396
 397        :return:
 398        """
 399
 400        yield self.env.timeout(timeout)
 401
 402        # If the selected zone is a compute one, sent to the manager zone that the vnfs will be placed in that zone.
 403        if zone.zone_type == Zone.TYPE_COMPUTE:
 404            self.vnf_segment_log.add_event(
 405                event=VNFSegmentLog.COMPUTE_ZONE_SELECTED,
 406                time=self.env.now,
 407                sfc_request_name=sfc_request.name,
 408                zone_name=zone.name,
 409                vnf_names=vnf_names
 410            )
 411
 412            zone_manager = self.sfc_request_zone_manager[sfc_request.name]
 413            ret = self.zdsm[zone_manager.name].add_segment_to_compute_zone(
 414                sfc_request=sfc_request,
 415                vnf_names=vnf_names,
 416                zone=zone
 417            )
 418
 419            # The computed zone were selected after the timeout
 420            if not ret:
 421                self.vnf_segment_log.add_event(
 422                    event=VNFSegmentLog.TIMEOUT,
 423                    time=self.env.now,
 424                    sfc_request_name=sfc_request.name,
 425                    zone_name=zone.name,
 426                    vnf_names=vnf_names
 427                )
 428            return
 429
 430        # If the selected zone is an aggregation zone it means that we need to execute the game again.
 431        if zone.zone_type == Zone.TYPE_AGGREGATION:
 432            # Log that the aggregation zone is selected to execute a set of VNFs
 433            self.vnf_segment_log.add_event(
 434                event=VNFSegmentLog.AGGREGATION_ZONE_SELECTED,
 435                time=self.env.now,
 436                sfc_request_name=sfc_request.name,
 437                zone_name=zone.name,
 438                vnf_names=vnf_names
 439            )
 440
 441        # Run the distributed placement to the other zones
 442        dsm: DistributedServiceManager = self.zdsm[zone.name]
 443
 444        plans = self.find_valid_vnf_segment_plan(
 445            zone=zone,
 446            vnf_names=vnf_names
 447        )
 448
 449        selected_segmentation_plan = dsm.speed.select_segmentation_plan(plans)
 450
 451        # This is the game implementation
 452        selected_child_zones = dsm.select_zones_to_vnf_segments(selected_segmentation_plan)
 453
 454        for child_zone_name, vnfs in selected_child_zones.items():
 455            cz = self.zones[child_zone_name]
 456
 457            timeout_to_child_zone = self.delay_between_distributed_service_components(
 458                zone_1=zone,
 459                zone_2=cz
 460            )
 461
 462            # Send in parallel the request to place each VNF Segment.
 463            self.env.process(
 464                self.distributed_sfc_placement_process(
 465                    sfc_request=sfc_request,
 466                    zone=cz,
 467                    vnf_names=vnfs['vnfs'],
 468                    timeout=timeout_to_child_zone
 469                )
 470            )
 471
 472        return

 474    def delay_between_distributed_service_components(self, zone_1: Zone, zone_2: Zone) -> int:
 475        """
 476        Return the delay between the distributed service components in two zones.
 477
 478        :param zone_1: The first zone.
 479        :param zone_2: The second zone.
 480        :return:
 481        """
 482        topo: Topology = self.environment['topology']
 483
 484        sp = topo.shortest_simple_edge_path(
 485            src_name=self.zdsm[zone_1.name].node.name,
 486            dst_name=self.zdsm[zone_2.name].node.name,
 487        )
 488
 489        timeout = topo.path_delay(path=sp)
 490
 491        return timeout

 493    def update_aggregated_data(self):
 494        """
 495        Update the aggregated data in all the zones. First the child zones will send the data to
 496        parent zone recursively.
 497
 498        :return:
 499        """
 500        zones_to_update = nx.dfs_tree(self.graph_zones)
 501        list_zones = list(zones_to_update.nodes())
 502
 503        # execute the update from the bottom to top
 504        list_zones.reverse()
 505
 506        for zone_name in list_zones:
 507            zone = self.zones[zone_name]
 508            self.update_zone_aggregated_data(
 509                zone=zone

 533    def select_zone_manager(self, sfc_request: SFCRequest) -> dict:
 534        """
 535        Select the zone that manage the distributed placement process.
 536
 537        The src and dst must be in one of the underlying zones.
 538
 539        :param sfc_request: The SFC Request
 540        :return:
 541        """
 542        # Update the aggregated data for all the zones.
 543        self.update_aggregated_data()
 544
 545        src_domain_name = self.domain_zone[sfc_request.src.domain_name]
 546        dst_domain_name = self.domain_zone[sfc_request.dst.domain_name]
 547
 548        graph = self.graph_zones
 549
 550        zone_manager_name = nx.lowest_common_ancestor(graph, src_domain_name, dst_domain_name)
 551
 552        zone_manager: Zone = self.zones[zone_manager_name]
 553
 554        vnf_names: List[str] = list()
 555        for vnf in sfc_request.sfc.vnfs:
 556            vnf_names.append(vnf.name)
 557
 558        valid_zone_manager = False
 559
 560        while True:
 561            valid_plans = self.find_valid_vnf_segment_plan(
 562                zone=zone_manager,
 563                vnf_names=vnf_names
 564            )
 565
 566            if valid_plans:
 567                valid_zone_manager = True
 568                break
 569
 570            if not valid_plans and not zone_manager.parent_zone_name:
 571                valid_zone_manager = False
 572                break
 573
 574            if not valid_plans and zone_manager.parent_zone_name:
 575                zone_manager: Zone = self.zones[zone_manager.parent_zone_name]
 576                zone_manager_name = zone_manager.name
 577
 578        if not valid_zone_manager:
 579            raise TypeError("The infrastructure can not execute the Service Requested {}.".format(sfc_request.name))
 580
 581        return {
 582            'zone_manager': zone_manager,
 583            'plans': valid_plans
 584        }

 605    def execute_placement_plan(self, domain: Domain, plan: SFCPlacementPlan):
 606        """
 607        Execute the SFC Placement plan. Create the SFC Instance, VNF Instances and Virtual Links.
 608
 609        :param domain: The domain object.
 610        :param plan: The SFC Placement plan.
 611        """
 612        # sdn_controller = domain.get_sdn_controller()
 613        sdn_controller = self.sdn_controller
 614
 615        sfc_request: SFCRequest = self.sfc_requests[plan['sfc_request_name']]
 616
 617        vnf_instances = domain.create_vnf_instance_from_placement_plan(
 618            plan=plan,
 619            initial_status=self.vnf_instance_initial_status,
 620            ru=self.vnf_instance_ru_status
 621        )
 622
 623        """
 624        For each created VNF Instance, a simpy.Resource object will be created. SimPy use this object to control the
 625        access to the resource. For example, if a packet needs to be processed in the VNF Instance but already
 626        exist a packet being processed, thus the second packet must wait until the resource became free.
 627        """
 628        aux_vnf_instances: Dict[str, VNFInstance] = dict()
 629        for vnf_instance in vnf_instances:
 630            aux_vnf_instances[vnf_instance.vnf.name] = vnf_instance
 631            max_share = 1
 632            if vnf_instance.extra_parameter_is_defined("max_share"):
 633                max_share = vnf_instance.get_extra_parameter("max_share")
 634
 635            self.simpy_resource_vnf_instances[vnf_instance.name] = simpy.Resource(self.env, capacity=max_share)
 636
 637            # Create the log from VNF Instance creation.
 638            self.vnf_instance_log.add_event(
 639                event=VNFInstanceLog.CREATED,
 640                time=self.env.now,
 641                domain_name=domain.name,
 642                vnf_instance=vnf_instance
 643            )
 644
 645        # Create the Virtual Links defined in the placement plan.
 646        aux_virtual_link: Dict[str, VirtualLink] = dict()
 647        for key, vnf in plan['vnfs'].items():
 648            vl_path = vnf['virtual_link_path']
 649
 650            # It means that the next VNF is in the same node, thus a virtual link will not be necessary.
 651            if not vl_path:
 652                continue
 653

 722    def workload_process(self, distributed_service: DistributedService):
 723        """
 724        Start processing the packets (workload) of the SFC Request.
 725
 726        :param distributed_service: The distributed service.
 727        """
 728
 729        sfc_request_name = distributed_service.sfc_request.name
 730
 731        packets: Dict[str, Packet] = self.packets[sfc_request_name]
 732        """
 733        All the packets of the SFC Request.
 734        """
 735
 736        # In which time each packet must be processed. Multiple packets can arrive at same time.
 737        packets_time: Dict[int, List[Packet]] = PacketHelper.compute_packet_execution_time_based_on_simulation_time(
 738            packets=packets,
 739            time=self.env.now
 740        )
 741
 742        # Run the packet workload
 743        while True:
 744            now = int(self.env.now)
 745            if now in packets_time.keys():
 746                packets_to_process = packets_time[now]
 747                for packet in packets_to_process:
 748                    # Packets will not wait to start being processed, thus the yield is unnecessary.
 749                    self.env.process(
 750                        self.process_packet(
 751                            packet=packet,
 752                            distributed_service=distributed_service
 753                        )
 754                    )
 755
 756            # Execute the simulation tick of 1ms
 757            yield self.env.timeout(1)

 759    def process_packet(self, packet: Packet, distributed_service: DistributedService):
 760        """
 761        Process the packets over the ingress link, VNF Instances and Virtual Links.
 762
 763        :param packet: The packet object.
 764        :param distributed_service: The Distributed Service.
 765        :return:
 766        """
 767

 769
 770        sfc_instance = execution_domain.get_sfc_instance(
 771            sfc_request_name=distributed_service.sfc_request.name
 772        )
 773
 774        sfc_request = distributed_service.sfc_request
 775
 776        sfc_instance.reset_timeout()

 778        A new packet to be processed will reset the SFC Instance timeout.
 779        """

 782        self.packet_log.add_event(
 783            event=PacketLog.CREATED,
 784            time=self.env.now,

 786            packet=packet
 787        )
 788
 789        self.packet_in_execution[sfc_request.src.domain_name].append(packet)
 790        """
 791        Add the packet in the list of packet that are in execution
 792        """
 793
 794        # Process the link the inter-domain link
 795        if distributed_service.ingress_link:
 796            yield self.env.process(
 797                self.process_packet_over_link(
 798                    packet=packet,
 799                    virtual_link=distributed_service.ingress_link,
 800                    domain=self.domains[sfc_request.src.domain_name]
 801                )
 802            )
 803
 804        # If the SFC Instance have an ingress link thus process the packet over it.
 805        if sfc_instance.ingress_link:
 806            yield self.env.process(
 807                self.process_packet_over_link(
 808                    packet=packet,
 809                    virtual_link=sfc_instance.ingress_link,
 810                    domain=execution_domain
 811                )
 812            )
 813
 814        for vnf in sfc_instance.sfc_request.sfc.vnfs:
 815            vnf_instance: VNFInstance = sfc_instance.vnf_instances[vnf.name]
 816            # Process the packet over the VNF Instance
 817            yield self.env.process(
 818                self.process_packet_over_vnf_instance(
 819                    packet=packet,
 820                    vnf_instance=vnf_instance,
 821                    domain=execution_domain
 822                )
 823            )
 824
 825            # Only process the time in the virtual link if it exists.
 826            if vnf.name in sfc_instance.virtual_links.keys():
 827                virtual_link: VirtualLink = sfc_instance.virtual_links[vnf.name]
 828                # Process the packet over the Virtual Link
 829                yield self.env.process(
 830                    self.process_packet_over_link(
 831                        packet=packet,
 832                        virtual_link=virtual_link,
 833                        domain=execution_domain
 834                    )
 835                )
 836
 837        # Process the link the inter-domain link
 838        if distributed_service.egress_link:
 839            yield self.env.process(
 840                self.process_packet_over_link(
 841                    packet=packet,
 842                    virtual_link=distributed_service.egress_link,
 843                    domain=self.domains[sfc_request.dst.domain_name]
 844                )
 845            )
 846
 847        # Log that the packet was processed
 848        self.packet_log.add_event(
 849            event=PacketLog.PROCESSED,
 850            time=self.env.now,
 851            domain_name=sfc_request.dst.domain_name,
 852            packet=packet
 853        )
 854
 855        self.packet_in_execution[sfc_request.src.domain_name].remove(packet)
 856        """
 857        Remove the packet from the list of packet that are in execution
 858        """

 257                    raise TypeError("The node_mem_default_cost is not defined in the simulation config file")

 259                if not node.extra_parameter_is_defined("cpu_cost"):
 260                    node.add_extra_parameter("cpu_cost", node_cpu_default_cost)
 261
 262                if not node.extra_parameter_is_defined("mem_cost"):
 263                    node.add_extra_parameter("mem_cost", node_mem_default_cost)
 264
 265            # Configure the default link cost for links without extra_parameters.
 266            for link_name, link in self.environment['links'].items():
 267                if 'link_default_cost' in self.config.keys():
 268                    link_default_cost = self.config['link_default_cost']
 269                else:
 270                    raise TypeError("The link_default_cost is not defined in the simulation config file")
 271
 272                if not link.extra_parameter_is_defined("cost"):
 273                    link.add_extra_parameter("cost", link_default_cost)

 275    def run(self):
 276        """
 277        Execute the SimPy simulation process.
 278
 279        :return: None
 280        """
 281        self.env.process(
 282            self.simulate()
 283        )
 284        """
 285        Add the main simulation process in the SimPy environment.

 298    def simulate(self):
 299        """
 300        Execute the simulation main process.
 301        """
 302
 303        while True:
 304
 305            # Process the SFC Requests that arrived in the simulation time (env.now)
 306            if self.env.now in self.environment['sfc_requests_arrival']:
 307                sfc_requests = self.environment['sfc_requests_arrival'][self.env.now]
 308
 309                # Execute the SPC Placement in a distributed fashion for each SFC Request.
 310                for sfc_request in sfc_requests:
 311
 312                    # Update all the aggregated data in the simulation
 313                    self.update_aggregated_data()
 314
 315                    try:
 316
 317                        # Select the zone manager
 318                        aux = self.select_zone_manager(
 319                            sfc_request=sfc_request
 320                        )
 321
 322                        zone_manager: Zone = aux['zone_manager']
 323
 324                        vnf_names = DistributedServiceHelper.get_vnf_names_from_sfc_request(
 325                            sfc_request=sfc_request
 326                        )
 327
 328                        # The zone that will manage this request
 329                        self.sfc_request_zone_manager[sfc_request.name] = zone_manager
 330
 331                        placement_timeout = self.default_placement_timeout
 332
 333                        if sfc_request.extra_parameter_is_defined('placement_timeout'):
 334                            placement_timeout = sfc_request.get_extra_parameter('placement_timeout')
 335
 336                        self.zdsm[zone_manager.name].add_sfc_request(
 337                            sfc_request=sfc_request,
 338                            placement_timeout=placement_timeout
 339                        )
 340
 341                        # Lot the selection zone will manage the SFC Request
 342                        self.distributed_service_log.add_event(
 343                            event=DistributedServiceLog.ZONE_MANAGER_SELECTED,
 344                            time=self.env.now,
 345                            sfc_request_name=sfc_request.name,
 346                            zone_manager_name=zone_manager.name
 347                        )
 348
 349                    except TypeError:
 350                        # The requested service cannot be placed, there is no zone to manage this request.
 351                        self.distributed_service_log.add_event(
 352                            event=DistributedServiceLog.FAIL,
 353                            time=self.env.now,
 354                            sfc_request_name=sfc_request.name,
 355                            zone_manager_name="Not Found"
 356                        )
 357                        continue
 358
 359                    # Zone manager executing the game
 360                    try:
 361                        self.env.process(
 362                            self.distributed_sfc_placement_process(
 363                                sfc_request=sfc_request,
 364                                zone=zone_manager,
 365                                vnf_names=vnf_names
 366                            )
 367                        )
 368                    except TypeError:
 369                        print("Simulation error")
 370
 371            for i, domain in self.environment['domains'].items():
 372                self.dec_sfc_instances_timeout(domain)
 373                """
 374                Decrement the timeout of the SFC Instances.
 375                """
 376
 377                self.inc_packet_delay(domain)
 378                """
 379                Increment the delay of the packet in execution.
 380                """
 381
 382            # Iterate over the SFC Requests and log when the plan for placement arrives to the manager zone.
 383            self.sfc_requests_vnfs_are_assigned_to_compute_zone()
 384
 385            # Do the simulation tick of 1ms
 386            yield self.env.timeout(1)

 388    def distributed_sfc_placement_process(self, sfc_request: SFCRequest, zone: Zone, vnf_names: List, timeout: int = 0):
 389        """
 390        Execute the game in each zone.
 391
 392        :param sfc_request: The service requested
 393        :param zone: The zone where the game is executed
 394        :param vnf_names: The name of the VNFs.
 395        :param timeout: The delay between the distributed service component in the parent until the child
 396        zone component.
 397
 398        :return:
 399        """
 400
 401        yield self.env.timeout(timeout)
 402
 403        # If the selected zone is a compute one, sent to the manager zone that the vnfs will be placed in that zone.
 404        if zone.zone_type == Zone.TYPE_COMPUTE:
 405            self.vnf_segment_log.add_event(
 406                event=VNFSegmentLog.COMPUTE_ZONE_SELECTED,
 407                time=self.env.now,
 408                sfc_request_name=sfc_request.name,
 409                zone_name=zone.name,
 410                vnf_names=vnf_names
 411            )
 412
 413            zone_manager = self.sfc_request_zone_manager[sfc_request.name]
 414            ret = self.zdsm[zone_manager.name].add_segment_to_compute_zone(
 415                sfc_request=sfc_request,
 416                vnf_names=vnf_names,
 417                zone=zone
 418            )
 419
 420            # The computed zone were selected after the timeout
 421            if not ret:
 422                self.vnf_segment_log.add_event(
 423                    event=VNFSegmentLog.TIMEOUT,
 424                    time=self.env.now,
 425                    sfc_request_name=sfc_request.name,
 426                    zone_name=zone.name,
 427                    vnf_names=vnf_names
 428                )
 429            return
 430
 431        # If the selected zone is an aggregation zone it means that we need to execute the game again.
 432        if zone.zone_type == Zone.TYPE_AGGREGATION:
 433            # Log that the aggregation zone is selected to execute a set of VNFs
 434            self.vnf_segment_log.add_event(
 435                event=VNFSegmentLog.AGGREGATION_ZONE_SELECTED,
 436                time=self.env.now,
 437                sfc_request_name=sfc_request.name,
 438                zone_name=zone.name,
 439                vnf_names=vnf_names
 440            )
 441
 442        # Run the distributed placement to the other zones
 443        dsm: DistributedServiceManager = self.zdsm[zone.name]
 444
 445        plans = self.find_valid_vnf_segment_plan(
 446            zone=zone,
 447            vnf_names=vnf_names
 448        )
 449
 450        selected_segmentation_plan = dsm.speed.select_segmentation_plan(plans)
 451
 452        # This is the game implementation
 453        selected_child_zones = dsm.select_zones_to_vnf_segments(selected_segmentation_plan)
 454
 455        for child_zone_name, vnfs in selected_child_zones.items():
 456            cz = self.zones[child_zone_name]
 457
 458            timeout_to_child_zone = self.delay_between_distributed_service_components(
 459                zone_1=zone,
 460                zone_2=cz
 461            )
 462
 463            # Send in parallel the request to place each VNF Segment.
 464            self.env.process(
 465                self.distributed_sfc_placement_process(
 466                    sfc_request=sfc_request,
 467                    zone=cz,
 468                    vnf_names=vnfs['vnfs'],
 469                    timeout=timeout_to_child_zone
 470                )
 471            )
 472
 473        return

 475    def delay_between_distributed_service_components(self, zone_1: Zone, zone_2: Zone) -> int:
 476        """
 477        Return the delay between the distributed service components in two zones.
 478
 479        :param zone_1: The first zone.
 480        :param zone_2: The second zone.
 481        :return:
 482        """
 483        topo: Topology = self.environment['topology']
 484
 485        sp = topo.shortest_simple_edge_path(
 486            src_name=self.zdsm[zone_1.name].node.name,
 487            dst_name=self.zdsm[zone_2.name].node.name,
 488        )
 489
 490        timeout = topo.path_delay(path=sp)
 491
 492        return timeout

 494    def update_aggregated_data(self):
 495        """
 496        Update the aggregated data in all the zones. First the child zones will send the data to
 497        parent zone recursively.
 498
 499        :return:
 500        """
 501        zones_to_update = nx.dfs_tree(self.graph_zones)
 502        list_zones = list(zones_to_update.nodes())
 503
 504        # execute the update from the bottom to top
 505        list_zones.reverse()
 506
 507        for zone_name in list_zones:
 508            zone = self.zones[zone_name]
 509            self.update_zone_aggregated_data(
 510                zone=zone
 511            )

 513    def update_zone_aggregated_data(self, zone: Zone):
 514        """
 515        Update the aggregated data for a zone.
 516
 517        :return:
 518        """
 519        if zone.zone_type == Zone.TYPE_ACCESS:
 520            return
 521
 522        dsm = self.zdsm[zone.name]
 523
 524        # Aggregate the data from its own child zone
 525        aggregated_data = dsm.speed.aggregate_date()
 526
 527        # Send data to the parent zone.
 528        if zone.parent_zone_name:
 529            self.zdsm[zone.parent_zone_name].speed.update_child_zone_aggregated_data(
 530                zone_name=zone.name,
 531                child_zone_aggregated_data=aggregated_data
 532            )

 534    def select_zone_manager(self, sfc_request: SFCRequest) -> dict:
 535        """
 536        Select the zone that manage the distributed placement process.
 537
 538        The src and dst must be in one of the underlying zones.
 539
 540        :param sfc_request: The SFC Request
 541        :return:
 542        """
 543        # Update the aggregated data for all the zones.
 544        self.update_aggregated_data()
 545
 546        src_domain_name = self.domain_zone[sfc_request.src.domain_name]
 547        dst_domain_name = self.domain_zone[sfc_request.dst.domain_name]
 548
 549        graph = self.graph_zones
 550
 551        zone_manager_name = nx.lowest_common_ancestor(graph, src_domain_name, dst_domain_name)
 552
 553        zone_manager: Zone = self.zones[zone_manager_name]
 554
 555        vnf_names: List[str] = list()
 556        for vnf in sfc_request.sfc.vnfs:
 557            vnf_names.append(vnf.name)
 558
 559        valid_zone_manager = False
 560
 561        while True:
 562            valid_plans = self.find_valid_vnf_segment_plan(
 563                zone=zone_manager,
 564                vnf_names=vnf_names
 565            )
 566
 567            if valid_plans:
 568                valid_zone_manager = True
 569                break
 570
 571            if not valid_plans and not zone_manager.parent_zone_name:
 572                valid_zone_manager = False
 573                break
 574
 575            if not valid_plans and zone_manager.parent_zone_name:
 576                zone_manager: Zone = self.zones[zone_manager.parent_zone_name]
 577                zone_manager_name = zone_manager.name
 578
 579        if not valid_zone_manager:
 580            raise TypeError("The infrastructure can not execute the Service Requested {}.".format(sfc_request.name))
 581
 582        return {
 583            'zone_manager': zone_manager,
 584            'plans': valid_plans
 585        }

 606    def execute_placement_plan(self, domain: Domain, plan: SFCPlacementPlan):
 607        """
 608        Execute the SFC Placement plan. Create the SFC Instance, VNF Instances and Virtual Links.
 609
 610        :param domain: The domain object.
 611        :param plan: The SFC Placement plan.
 612        """
 613        # sdn_controller = domain.get_sdn_controller()
 614        sdn_controller = self.sdn_controller
 615
 616        sfc_request: SFCRequest = self.sfc_requests[plan['sfc_request_name']]
 617
 618        vnf_instances = domain.create_vnf_instance_from_placement_plan(
 619            plan=plan,
 620            initial_status=self.vnf_instance_initial_status,
 621            ru=self.vnf_instance_ru_status
 622        )
 623
 624        """
 625        For each created VNF Instance, a simpy.Resource object will be created. SimPy use this object to control the
 626        access to the resource. For example, if a packet needs to be processed in the VNF Instance but already
 627        exist a packet being processed, thus the second packet must wait until the resource became free.
 628        """
 629        aux_vnf_instances: Dict[str, VNFInstance] = dict()
 630        for vnf_instance in vnf_instances:
 631            aux_vnf_instances[vnf_instance.vnf.name] = vnf_instance
 632            max_share = 1
 633            if vnf_instance.extra_parameter_is_defined("max_share"):
 634                max_share = vnf_instance.get_extra_parameter("max_share")
 635
 636            self.simpy_resource_vnf_instances[vnf_instance.name] = simpy.Resource(self.env, capacity=max_share)
 637
 638            # Create the log from VNF Instance creation.
 639            self.vnf_instance_log.add_event(
 640                event=VNFInstanceLog.CREATED,
 641                time=self.env.now,
 642                domain_name=domain.name,
 643                vnf_instance=vnf_instance
 644            )
 645
 646        # Create the Virtual Links defined in the placement plan.
 647        aux_virtual_link: Dict[str, VirtualLink] = dict()
 648        for key, vnf in plan['vnfs'].items():
 649            vl_path = vnf['virtual_link_path']
 650
 651            # It means that the next VNF is in the same node, thus a virtual link will not be necessary.
 652            if not vl_path:
 653                continue
 654
 655            virtual_link = sdn_controller.create_virtual_link_from_path(
 656                path=vl_path,
 657                bw=sfc_request.sfc.required_bw
 658            )
 659
 660            aux_virtual_link[key] = virtual_link
 661
 662            self.virtual_link_log.add_event(
 663                event=VirtualLinkLog.CREATED,
 664                time=self.env.now,
 665                domain_name=domain.name,
 666                virtual_link=virtual_link
 667            )
 668
 669            self.simpy_resource_virtual_links[virtual_link.name] = simpy.Resource(self.env)
 670
 671        # Create the virtual link to the ingress node.
 672        ingress_link = None
 673        if plan['ingress_link_path']:
 674            ingress_link = sdn_controller.create_virtual_link_from_path(
 675                path=plan['ingress_link_path'],
 676                bw=sfc_request.sfc.required_bw
 677            )
 678
 679            self.virtual_link_log.add_event(
 680                event=VirtualLinkLog.CREATED,
 681                time=self.env.now,
 682                domain_name=domain.name,
 683                virtual_link=ingress_link
 684            )
 685
 686            self.simpy_resource_virtual_links[ingress_link.name] = simpy.Resource(self.env)
 687
 688        sfc_instance = domain.create_sfc_instance(
 689            sfc_request=sfc_request,
 690            vnf_instances=aux_vnf_instances,
 691            virtual_links=aux_virtual_link,
 692            ingress_link=ingress_link
 693        )
 694
 695        if sfc_instance:
 696
 697            # Log the success in the placement plan creation.
 698            self.placement_log.add_event(
 699                event=PlacementLog.SUCCESS,
 700                time=self.env.now,
 701                domain_name=domain.name,
 702                sfc_request=sfc_request
 703            )
 704
 705            self.sfc_instance_log.add_event(
 706                event=SFCInstanceLog.CREATED,
 707                time=self.env.now,
 708                domain_name=domain.name,
 709                sfc_instance=sfc_instance
 710            )
 711            return sfc_instance
 712        else:
 713            # Log the success in the placement plan creation.
 714            self.placement_log.add_event(
 715                event=PlacementLog.FAIL,
 716                time=self.env.now,
 717                domain_name=domain.name,
 718                sfc_request=sfc_request
 719            )
 720
 721            raise TypeError("The SFC Instance could not be created.")

 723    def workload_process(self, distributed_service: DistributedService):
 724        """
 725        Start processing the packets (workload) of the SFC Request.
 726
 727        :param distributed_service: The distributed service.
 728        """
 729
 730        sfc_request_name = distributed_service.sfc_request.name
 731
 732        packets: Dict[str, Packet] = self.packets[sfc_request_name]
 733        """
 734        All the packets of the SFC Request.
 735        """
 736
 737        # In which time each packet must be processed. Multiple packets can arrive at same time.
 738        packets_time: Dict[int, List[Packet]] = PacketHelper.compute_packet_execution_time_based_on_simulation_time(
 739            packets=packets,
 740            time=self.env.now
 741        )
 742
 743        # Run the packet workload
 744        while True:
 745            now = int(self.env.now)
 746            if now in packets_time.keys():
 747                packets_to_process = packets_time[now]
 748                for packet in packets_to_process:
 749                    # Packets will not wait to start being processed, thus the yield is unnecessary.
 750                    self.env.process(
 751                        self.process_packet(
 752                            packet=packet,
 753                            distributed_service=distributed_service
 754                        )
 755                    )
 756
 757            # Execute the simulation tick of 1ms
 758            yield self.env.timeout(1)

 760    def process_packet(self, packet: Packet, distributed_service: DistributedService):
 761        """
 762        Process the packets over the ingress link, VNF Instances and Virtual Links.
 763
 764        :param packet: The packet object.
 765        :param distributed_service: The Distributed Service.
 766        :return:
 767        """
 768
 769        execution_domain: Domain = self.domains[distributed_service.execution_domain_name]
 770
 771        sfc_instance = execution_domain.get_sfc_instance(
 772            sfc_request_name=distributed_service.sfc_request.name
 773        )
 774
 775        sfc_request = distributed_service.sfc_request
 776
 777        sfc_instance.reset_timeout()
 778        """
 779        A new packet to be processed will reset the SFC Instance timeout.
 780        """
 781
 782        # Log the packet creation
 783        self.packet_log.add_event(
 784            event=PacketLog.CREATED,
 785            time=self.env.now,
 786            domain_name=sfc_request.src.domain_name,
 787            packet=packet
 788        )
 789
 790        self.packet_in_execution[sfc_request.src.domain_name].append(packet)
 791        """
 792        Add the packet in the list of packet that are in execution
 793        """
 794
 795        # Process the link the inter-domain link
 796        if distributed_service.ingress_link:
 797            yield self.env.process(
 798                self.process_packet_over_link(
 799                    packet=packet,
 800                    virtual_link=distributed_service.ingress_link,
 801                    domain=self.domains[sfc_request.src.domain_name]
 802                )
 803            )
 804
 805        # If the SFC Instance have an ingress link thus process the packet over it.
 806        if sfc_instance.ingress_link:
 807            yield self.env.process(
 808                self.process_packet_over_link(
 809                    packet=packet,
 810                    virtual_link=sfc_instance.ingress_link,
 811                    domain=execution_domain
 812                )
 813            )
 814
 815        for vnf in sfc_instance.sfc_request.sfc.vnfs:
 816            vnf_instance: VNFInstance = sfc_instance.vnf_instances[vnf.name]
 817            # Process the packet over the VNF Instance
 818            yield self.env.process(
 819                self.process_packet_over_vnf_instance(
 820                    packet=packet,
 821                    vnf_instance=vnf_instance,
 822                    domain=execution_domain
 823                )
 824            )
 825
 826            # Only process the time in the virtual link if it exists.
 827            if vnf.name in sfc_instance.virtual_links.keys():
 828                virtual_link: VirtualLink = sfc_instance.virtual_links[vnf.name]
 829                # Process the packet over the Virtual Link
 830                yield self.env.process(
 831                    self.process_packet_over_link(
 832                        packet=packet,
 833                        virtual_link=virtual_link,
 834                        domain=execution_domain
 835                    )
 836                )
 837
 838        # Process the link the inter-domain link
 839        if distributed_service.egress_link:
 840            yield self.env.process(
 841                self.process_packet_over_link(
 842                    packet=packet,
 843                    virtual_link=distributed_service.egress_link,
 844                    domain=self.domains[sfc_request.dst.domain_name]
 845                )
 846            )
 847
 848        # Log that the packet was processed
 849        self.packet_log.add_event(
 850            event=PacketLog.PROCESSED,
 851            time=self.env.now,
 852            domain_name=sfc_request.dst.domain_name,
 853            packet=packet
 854        )
 855
 856        self.packet_in_execution[sfc_request.src.domain_name].remove(packet)
 857        """
 858        Remove the packet from the list of packet that are in execution
 859        """
 860
 861    def process_packet_over_vnf_instance(self, packet: Packet, vnf_instance: VNFInstance, domain: Domain):
 862        """
 863        Process the packet over the VNF Instance.
 864
 865        :param packet: The packet object.
 866        :param vnf_instance: The VNF Instance object.
 867        :param domain: The domain object.
 868        :return:
 869        """
 870        domain_name = domain.name
 871        resource_vnf_instance = self.simpy_resource_vnf_instances[vnf_instance.name]
 872
 873        self.packet_log.add_event(
 874            event=PacketLog.PACKET_VNF_INSTANCE_ARRIVED,
 875            time=self.env.now,
 876            domain_name=domain_name,
 877            vnf_instance=vnf_instance,
 878            packet=packet
 879        )
 880
 881        with resource_vnf_instance.request() as request:
 882            yield request
 883
 884            self.packet_log.add_event(
 885                event=PacketLog.PACKET_VNF_INSTANCE_ARRIVED_START_PROCESSING,
 886                time=self.env.now,
 887                domain_name=domain_name,
 888                vnf_instance=vnf_instance,
 889                packet=packet
 890            )
 891
 892            process_delay = vnf_instance.process_delay(packet.cpu_demand)
 893
 894            yield self.env.timeout(process_delay)
 895
 896            self.packet_log.add_event(
 897                event=PacketLog.PACKET_VNF_INSTANCE_PROCESSED,
 898                time=self.env.now,
 899                domain_name=domain_name,
 900                vnf_instance=vnf_instance,
 901                packet=packet
 902            )
 903
 904        return True

 906    def process_packet_over_link(self, packet: Packet, virtual_link: VirtualLink, domain: Domain):
 907        """
 908        Process the packet over the Virtual link.
 909
 910        :param packet: The packet object.
 911        :param virtual_link: The virtual link object.
 912        :param domain: The domain object
 913        :return:
 914        """
 915        domain_name = domain.name
 916        resource_link = self.simpy_resource_virtual_links[virtual_link.name]
 917
 918        self.packet_log.add_event(
 919            event=PacketLog.PACKET_VIRTUAL_LINK_ARRIVED,
 920            time=self.env.now,
 921            domain_name=domain_name,
 922            virtual_link=virtual_link,
 923            packet=packet
 924        )
 925
 926        with resource_link.request() as request:
 927            yield request
 928
 929            self.packet_log.add_event(
 930                event=PacketLog.PACKET_VIRTUAL_LINK_START_PROCESSING,
 931                time=self.env.now,
 932                domain_name=domain_name,
 933                virtual_link=virtual_link,
 934                packet=packet
